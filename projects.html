-- 用户购买行为分析
SELECT
    user_id,
    COUNT(DISTINCT order_id) as order_count,
    SUM(amount) as total_amount,
    AVG(amount) as avg_amount
FROM user_orders
GROUP BY user_id
HAVING COUNT(DISTINCT order_id) > 5
ORDER BY total_amount DESC;

# 库存优化算法
 def optimize_inventory(forecast, lead_time, service_level=0.95):
     # 计算安全库存
     z_score = stats.norm.ppf(service_level)
     forecast_errors = np.abs(forecast - actual_sales)
     rmse = np.sqrt(np.mean(forecast_errors**2))
     safety_stock = z_score * rmse * np.sqrt(lead_time)

     # 计算经济订购数量 (EOQ)
     annual_demand = np.sum(forecast) * (365 / len(forecast))
     order_cost = 50  # 每次订购成本
     holding_cost = 0.2  # 库存持有成本率
     unit_cost = 10  # 单位成本

     eoq = np.sqrt((2 * annual_demand * order_cost) / (holding_cost * unit_cost))

     # 计算再订购点 (ROP)
     daily_demand = annual_demand / 365
     rop = daily_demand * lead_time + safety_stock

     return {
         'safety_stock': safety_stock,
         'eoq': eoq,
         'reorder_point': rop
     }
